@echo off
set app_url=http://localhost:8080/actuator/health # Replace with your application's health check URL
set retry_count=20

:retry
curl -s -o nul -I %app_url%
if %errorlevel% neq 0 (
    echo Application is down
    set /a retry_count-=1
    if %retry_count% gtr 0 (
        echo Retrying in 15 seconds...
        timeout /t 15
        goto retry
    ) else (
        echo Max retry count reached
    )
) else (
    echo Application is up and running
)

pause
